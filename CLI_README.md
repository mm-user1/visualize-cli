### Python —Å–∫—Ä–∏–ø—Ç-–æ–±–µ—Ä—Ç–∫–∞:
```python
import subprocess
import sys

def visualize(csv_in, csv_out, top=5, dates=None):
    cmd = [
        sys.executable, 'visualize_cli.py',
        '--csv_in', csv_in,
        '--csv_out', csv_out,
        '--top', str(top)
    ]
    
    if dates:
        cmd.extend(['--dates', dates])
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    print(result.stdout)
    
    if result.returncode != 0:
        print(f"Error: {result.stderr}", file=sys.stderr)
```
